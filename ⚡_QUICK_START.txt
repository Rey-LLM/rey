КОД ДЛЯ DASHBOARD (скопируй в client/src/pages/Dashboard.js)
═════════════════════════════════════════════════════════════════════════════

```javascript
import React, { useState, useEffect } from 'react';
import { useAuth } from '../AuthContext';
import axios from 'axios';

export default function Dashboard() {
  const { user } = useAuth();
  const [stats, setStats] = useState({
    activeProjects: 0,
    completedTasks: 0,
    focusHours: 0,
    breaksTaken: 0
  });

  useEffect(() => {
    fetchStats();
  }, [user]);

  const fetchStats = async () => {
    try {
      // Получить проекты
      const projectsRes = await axios.get('http://localhost:5000/api/projects', {
        headers: { Authorization: `Bearer ${localStorage.getItem('token')}` }
      });
      
      // Получить задачи
      const tasksRes = await axios.get('http://localhost:5000/api/tasks', {
        headers: { Authorization: `Bearer ${localStorage.getItem('token')}` }
      });

      const completedTasks = tasksRes.data?.filter(t => t.status === 'completed')?.length || 0;
      const breaks = JSON.parse(localStorage.getItem('breakReminderEnabled') || 'false') ? 4 : 0;

      setStats({
        activeProjects: projectsRes.data?.length || 0,
        completedTasks: completedTasks,
        focusHours: 6, // Пока просто число
        breaksTaken: breaks
      });
    } catch (error) {
      console.error('Error fetching stats:', error);
    }
  };

  return (
    <div style={{ padding: '2rem', maxWidth: '1200px', margin: '0 auto' }}>
      <h1 style={{ marginBottom: '2rem' }}>📊 Welcome Back, {user?.username}!</h1>
      
      {/* МЕТРИКИ */}
      <div style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(auto-fit, minmax(250px, 1fr))',
        gap: '1.5rem',
        marginBottom: '2rem'
      }}>
        <MetricCard 
          title="Active Projects" 
          value={stats.activeProjects} 
          icon="📁"
          color="#FF6B6B"
        />
        <MetricCard 
          title="Completed Tasks" 
          value={stats.completedTasks} 
          icon="✅"
          color="#4ECDC4"
        />
        <MetricCard 
          title="Focus Hours" 
          value={stats.focusHours} 
          icon="⏰"
          color="#FFD93D"
        />
        <MetricCard 
          title="Breaks Taken" 
          value={stats.breaksTaken} 
          icon="☕"
          color="#6BCB77"
        />
      </div>

      {/* ПРОДУКТИВНОСТЬ НА ЭТОЙ НЕДЕЛЕ */}
      <div style={{
        background: '#f5f5f5',
        padding: '2rem',
        borderRadius: '12px',
        marginBottom: '2rem'
      }}>
        <h2 style={{ marginTop: 0 }}>📈 Productivity This Week</h2>
        
        <div style={{ display: 'flex', gap: '1rem', alignItems: 'flex-end', height: '200px' }}>
          <ChartBar day="Mon" hours={8} maxHours={14} />
          <ChartBar day="Tue" hours={12} maxHours={14} />
          <ChartBar day="Wed" hours={9} maxHours={14} />
          <ChartBar day="Thu" hours={11} maxHours={14} />
          <ChartBar day="Fri" hours={7} maxHours={14} />
          <ChartBar day="Sat" hours={3} maxHours={14} />
          <ChartBar day="Sun" hours={2} maxHours={14} />
        </div>
      </div>

      {/* РЕКОМЕНДАЦИИ */}
      <div style={{
        background: 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)',
        color: 'white',
        padding: '2rem',
        borderRadius: '12px',
        marginBottom: '2rem'
      }}>
        <h2 style={{ marginTop: 0 }}>💡 AI Recommendations</h2>
        
        <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit, minmax(200px, 1fr))', gap: '1rem' }}>
          <Recommendation 
            emoji="☕"
            text="You've worked 6 hours. Take a break!"
          />
          <Recommendation 
            emoji="🎯"
            text="Best time to code: 10 AM - 12 PM"
          />
          <Recommendation 
            emoji="🔥"
            text="You're 12% more productive this week!"
          />
          <Recommendation 
            emoji="💪"
            text="Complete 2 more tasks for new badge"
          />
        </div>
      </div>

      {/* БЫСТРЫЕ ДЕЙСТВИЯ */}
      <div style={{
        background: '#f5f5f5',
        padding: '2rem',
        borderRadius: '12px'
      }}>
        <h2 style={{ marginTop: 0 }}>⚡ Quick Actions</h2>
        
        <div style={{ display: 'flex', gap: '1rem', flexWrap: 'wrap' }}>
          <button style={buttonStyle('primary')}>
            ➕ New Project
          </button>
          <button style={buttonStyle('secondary')}>
            ✅ New Task
          </button>
          <button style={buttonStyle('secondary')}>
            📄 View Documents
          </button>
          <button style={buttonStyle('secondary')}>
            🎯 Focus Mode
          </button>
        </div>
      </div>
    </div>
  );
}

// Компонент для карточки метрики
function MetricCard({ title, value, icon, color }) {
  return (
    <div style={{
      background: 'white',
      border: `3px solid ${color}`,
      padding: '1.5rem',
      borderRadius: '12px',
      textAlign: 'center',
      boxShadow: '0 2px 8px rgba(0,0,0,0.1)'
    }}>
      <div style={{ fontSize: '2.5rem', marginBottom: '0.5rem' }}>
        {icon}
      </div>
      <div style={{ fontSize: '0.9rem', color: '#666', marginBottom: '0.5rem' }}>
        {title}
      </div>
      <div style={{ fontSize: '2.5rem', fontWeight: 'bold', color }}>
        {value}
      </div>
    </div>
  );
}

// График столбец
function ChartBar({ day, hours, maxHours }) {
  const percentage = (hours / maxHours) * 100;
  
  return (
    <div style={{ flex: 1, display: 'flex', flexDirection: 'column', alignItems: 'center' }}>
      <div style={{
        width: '100%',
        height: percentage + '%',
        background: 'linear-gradient(180deg, #667eea 0%, #764ba2 100%)',
        borderRadius: '6px 6px 0 0',
        minHeight: '20px',
        transition: 'all 0.3s'
      }} />
      <div style={{ fontSize: '0.85rem', marginTop: '0.5rem', color: '#666' }}>
        {day}
      </div>
      <div style={{ fontSize: '0.75rem', color: '#999' }}>
        {hours}h
      </div>
    </div>
  );
}

// Рекомендация
function Recommendation({ emoji, text }) {
  return (
    <div style={{
      background: 'rgba(255,255,255,0.1)',
      padding: '1rem',
      borderRadius: '8px',
      textAlign: 'center'
    }}>
      <div style={{ fontSize: '2rem', marginBottom: '0.5rem' }}>
        {emoji}
      </div>
      <div style={{ fontSize: '0.95rem' }}>
        {text}
      </div>
    </div>
  );
}

// Стиль кнопок
function buttonStyle(type) {
  return {
    padding: '0.75rem 1.5rem',
    fontSize: '0.95rem',
    border: 'none',
    borderRadius: '8px',
    cursor: 'pointer',
    fontWeight: 'bold',
    transition: 'all 0.3s',
    background: type === 'primary' ? '#667eea' : '#e0e0e0',
    color: type === 'primary' ? 'white' : '#333',
    hover: {
      transform: 'translateY(-2px)',
      boxShadow: '0 4px 12px rgba(0,0,0,0.2)'
    }
  };
}
```
