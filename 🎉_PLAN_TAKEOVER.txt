```javascript
import React, { useState, useEffect } from 'react';
import { LineChart, Line, XAxis, YAxis, CartesianGrid } from 'recharts';

export default function Dashboard() {
  const [stats, setStats] = useState({
    activeProjects: 0,
    completedTasks: 0,
    focusHours: 0,
    breaksTaken: 0
  });

  useEffect(() => {
    // Загрузить статистику с сервера
    fetchStats();
  }, []);

  return (
    <div style={{ padding: '2rem' }}>
      <h1>📊 Your Dashboard</h1>
      
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 1fr)', gap: '1rem' }}>
        <MetricCard title="Active Projects" value={stats.activeProjects} icon="📁" />
        <MetricCard title="Completed Tasks" value={stats.completedTasks} icon="✅" />
        <MetricCard title="Focus Hours" value={stats.focusHours} icon="⏰" />
        <MetricCard title="Breaks Taken" value={stats.breaksTaken} icon="☕" />
      </div>
      
      <div style={{ marginTop: '2rem' }}>
        <h2>Productivity This Week</h2>
        <ProductivityChart />
      </div>
    </div>
  );
}

function MetricCard({ title, value, icon }) {
  return (
    <div style={{
      background: '#f5f5f5',
      padding: '1.5rem',
      borderRadius: '8px',
      textAlign: 'center'
    }}>
      <div style={{ fontSize: '2rem' }}>{icon}</div>
      <div style={{ fontSize: '0.9rem', color: '#666' }}>{title}</div>
      <div style={{ fontSize: '2rem', fontWeight: 'bold' }}>{value}</div>
    </div>
  );
}

function ProductivityChart() {
  const data = [
    { day: 'Mon', hours: 8 },
    { day: 'Tue', hours: 12 },
    { day: 'Wed', hours: 9 },
    { day: 'Thu', hours: 11 },
    { day: 'Fri', hours: 7 }
  ];

  return (
    <LineChart width={600} height={300} data={data}>
      <CartesianGrid strokeDasharray="3 3" />
      <XAxis dataKey="day" />
      <YAxis />
      <Line type="monotone" dataKey="hours" stroke="#8884d8" />
    </LineChart>
  );
}
```

2. Обновите App.js:

```javascript
import Dashboard from './pages/Dashboard';

// В Route добавьте:
<Route path="/dashboard" element={
  <PrivateRoute>
    <Dashboard />
  </PrivateRoute>
} />
```

3. Добавьте в Navbar ссылку на Dashboard
